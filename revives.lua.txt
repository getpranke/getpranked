--// Revive Duper Script by upio (Modified for ReplicatedStorage comms)
--// Method found by lolcat

--// Services
local Players = game:GetService("Players")
local StarterGui = game:GetService("StarterGui")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local VirtualInputManager = Instance.new("VirtualInputManager")

--// Remotes
local RemotesFolder = ReplicatedStorage:FindFirstChild("RemotesFolder")
if not RemotesFolder then
    -- Fallback if the game uses a different structure – adjust path as needed
    RemotesFolder = ReplicatedStorage:FindFirstChildWhichIsA("Folder") or ReplicatedStorage
end
local ReviveFriendEvent = RemotesFolder:FindFirstChild("ReviveFriend")
local ObtainReviveEvent = RemotesFolder:FindFirstChild("ObtainGiftedRevive")

--// Player Variables
local LocalPlayer = Players.LocalPlayer
local OtherPlayer

--// Game Data
local LatestRoom = ReplicatedStorage:FindFirstChild("GameData") and ReplicatedStorage.GameData:FindFirstChild("LatestRoom")
local Revives = LocalPlayer.PlayerGui:FindFirstChild("TopbarUI") and 
                LocalPlayer.PlayerGui.TopbarUI.Topbar.StatsTopbarHandler.StatModules.Revives.RevivesVal

--// Constants
local IsMainAccount = LocalPlayer.Name == AccountToDuplicateTo  -- must be defined before running
local DuplicationCount = DuplicationAmount or 1000
local Title = IsMainAccount and "Revive Dupe Helper (Main Account)" or "Revive Dupe Helper (Alt Account)"
local PacketPrefix = "ReviveDupe_"

--// Dupe State
local IsOtherAccountInitialized = false
local IsGiftingRevive = false

--// ============================
--// ReplicatedStorage Communication (replaces TextChatService)
--// ============================

-- Create a folder for cross‑client messages (if it doesn't exist)
local CommsFolder = ReplicatedStorage:FindFirstChild("Comms")
if not CommsFolder then
    CommsFolder = Instance.new("Folder")
    CommsFolder.Name = "Comms"
    CommsFolder.Parent = ReplicatedStorage
end

-- Each player gets their own StringValue inside the Comms folder
local MySignal = Instance.new("StringValue")
MySignal.Name = tostring(LocalPlayer.UserId)
MySignal.Parent = CommsFolder
MySignal.Value = ""  -- initial empty

-- Send a packet by writing to our own signal
local function SendPacket(packetName)
    local packetData = PacketPrefix .. packetName
    MySignal.Value = packetData
    -- Clear it after a short delay so we can send again immediately if needed
    task.defer(function()
        task.wait(0.1)
        if MySignal.Value == packetData then
            MySignal.Value = ""
        end
    end)
end

-- Listen for messages from any other player
local function setupMessageListener()
    for _, child in ipairs(CommsFolder:GetChildren()) do
        if child:IsA("StringValue") and child.Name ~= tostring(LocalPlayer.UserId) then
            local otherUserId = tonumber(child.Name)
            local connection
            connection = child.Changed:Connect(function(newValue)
                if newValue:sub(1, #PacketPrefix) == PacketPrefix then
                    local packetName = newValue:sub(#PacketPrefix + 1)
                    local sender = Players:GetPlayerByUserId(otherUserId)
                    if not sender then return end

                    -- Process packet (same logic as original)
                    if packetName == "Init" then
                        if IsOtherAccountInitialized then return end
                        IsOtherAccountInitialized = true
                        OtherPlayer = sender
                        SendPacket("Init")  -- acknowledge
                        StarterGui:SetCore("SendNotification", {
                            Title = Title,
                            Text = string.format("%s initialized, starting dupe process...", sender.Name),
                            Duration = 5
                        })
                    end

                    if not IsOtherAccountInitialized then return end

                    if packetName == "SendReviveStandardToMe" then
                        IsGiftingRevive = true
                        if OtherPlayer and OtherPlayer:GetAttribute("Alive") == true then
                            StarterGui:SetCore("SendNotification", {
                                Title = Title,
                                Text = "Waiting for the other account to die...",
                                Duration = 5
                            })
                            OtherPlayer:GetAttributeChangedSignal("Alive"):Wait()
                        end
                        task.wait(2.5)
                        if ReviveFriendEvent then
                            ReviveFriendEvent:FireServer(OtherPlayer.Name)
                        end
                        StarterGui:SetCore("SendNotification", {
                            Title = Title,
                            Text = "Revive sent to alt account. Now rejoin a new game with your alt account.",
                            Duration = 5
                        })
                    end
                end
            end)
        end
    end
end

-- Also listen for new players joining (they will create their StringValue later)
CommsFolder.ChildAdded:Connect(function(child)
    if child:IsA("StringValue") and child.Name ~= tostring(LocalPlayer.UserId) then
        setupMessageListener()  -- re-scan and attach listeners
    end
end)

-- Initial scan
setupMessageListener()

--// Send initial handshake packet
SendPacket("Init")

StarterGui:SetCore("SendNotification", {
    Title = Title,
    Text = "Waiting for other account to initialize...",
    Duration = 5
})

-- Wait for the other account to be detected
repeat task.wait() until IsOtherAccountInitialized

--// ============================
--// Helper Functions
--// ============================

local function AttemptToKillLocalPlayer()
    if LatestRoom and LatestRoom.Value == 0 then
        StarterGui:SetCore("SendNotification", {
            Title = "Revive Dupe Helper",
            Text = "Please open a door",
            Duration = 5
        })
        if LatestRoom then
            LatestRoom:GetPropertyChangedSignal("Value"):Wait()
        end
    end

    if replicatesignal then
        replicatesignal(LocalPlayer.Kill)
    else
        StarterGui:SetCore("SendNotification", {
            Title = "Revive Dupe Helper",
            Text = "Your executor does not support replicatesignal, please die manually",
            Duration = 5
        })
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
            LocalPlayer.Character.Humanoid.Died:Wait()
        end
    end

    StarterGui:SetCore("SendNotification", {
        Title = "Revive Dupe Helper",
        Text = "Make your alt account try and revive you. (Click more than once)",
        Duration = 5
    })
end

--// ============================
--// Main Logic
--// ============================

--// Gifting 1 revive to alt account process
if Revives and Revives.Value == 0 and not IsMainAccount then
    SendPacket("SendReviveStandardToMe")
    IsGiftingRevive = true

    AttemptToKillLocalPlayer()

    StarterGui:SetCore("SendNotification", {
        Title = Title,
        Text = "Waiting for the other account to gift you a revive...",
        Duration = 5
    })

    local AcceptReviveThread = task.spawn(function()
        while task.wait() do
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Return, false, game)
        end
    end)

    if Revives then
        Revives:GetPropertyChangedSignal("Value"):Wait()
    end
    coroutine.close(AcceptReviveThread)
    task.wait(0.05)
    game:Shutdown()
    return
end

--// Allow communication to settle
StarterGui:SetCore("SendNotification", {
    Title = Title,
    Text = "Please wait for 5 seconds to ensure proper communication has been established.",
    Duration = 5
})
task.wait(5)

if IsGiftingRevive then
    return
end

--// Main account duplication logic
if IsMainAccount then
    local ReviveObtainedAmount = 0
    local function OnObtainRevive(...)
        ReviveObtainedAmount = ReviveObtainedAmount + 1
        if ReviveObtainedAmount > (DuplicationCount * 0.85) then
            return true
        end
        task.wait(9e9)
        return true
    end

    if hookmetamethod then
        local mtHook; mtHook = hookmetamethod(game, "__newindex", function(self, key, value)
            if rawequal(self, ObtainReviveEvent) and key == "OnClientInvoke" then
                if not checkcaller() then
                    return
                end
            end
            return mtHook(self, key, value)
        end)
    else
        task.defer(function()
            while task.wait() do
                if ObtainReviveEvent then
                    ObtainReviveEvent.OnClientInvoke = OnObtainRevive
                end
            end
        end)
    end

    if ObtainReviveEvent then
        ObtainReviveEvent.OnClientInvoke = OnObtainRevive
    end

    AttemptToKillLocalPlayer()
else
    -- Alt account: wait for main to die, then spam revives
    if OtherPlayer and OtherPlayer:GetAttribute("Alive") then
        StarterGui:SetCore("SendNotification", {
            Title = Title,
            Text = "Waiting for the other account to die...",
            Duration = 5
        })
        OtherPlayer:GetAttributeChangedSignal("Alive"):Wait()
    end

    for i = 1, 5 do
        StarterGui:SetCore("SendNotification", {
            Title = Title,
            Text = string.format("Duping in %d seconds...", 6 - i),
            Duration = 1
        })
        task.wait(1)
    end

    for i = 1, DuplicationCount do
        if ReviveFriendEvent then
            ReviveFriendEvent:FireServer(OtherPlayer.Name)
        end
    end

    StarterGui:SetCore("SendNotification", {
        Title = Title,
        Text = "Duping completed!",
        Duration = 5
    })
end
